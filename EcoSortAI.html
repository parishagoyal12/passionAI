    <div class="container mx-auto p-8 rounded-xl shadow-xl bg-white text-center">

        <h1 class="text-3xl md:text-4xl font-bold mb-4">EcoSort AI</h1>

        <p class="text-gray-600 mb-6">Classify recyclable materials with AI.</p>

        

        <div class="mb-6">

            <input type="file" id="image-upload" accept="image/*" class="hidden">

            <label for="image-upload" class="cursor-pointer inline-block py-3 px-6 bg-green-500 hover:bg-green-600 transition text-white font-semibold rounded-full shadow-lg">

                Upload Image

            </label>

        </div>


        <div id="image-container" class="mb-6 hidden">

            <img id="image-preview" src="#" alt="Image Preview" class="image-preview mx-auto">

        </div>


        <button id="classify-btn" class="hidden py-3 px-6 bg-gray-800 hover:bg-gray-700 transition text-white font-semibold rounded-full shadow-lg">

            Classify

        </button>

        

        <div id="loading" class="mt-6 hidden text-gray-500 font-semibold">

            Analyzing...

        </div>


        <div id="result" class="mt-6 hidden">

            <h2 class="text-2xl font-bold mb-2">Result:</h2>

            <p id="prediction-text" class="text-xl font-semibold text-green-700"></p>

        </div>

    </div>


    <script>

        const imageUpload = document.getElementById('image-upload');

        const imageContainer = document.getElementById('image-container');

        const imagePreview = document.getElementById('image-preview');

        const classifyBtn = document.getElementById('classify-btn');

        const loadingText = document.getElementById('loading');

        const resultDiv = document.getElementById('result');

        const predictionText = document.getElementById('prediction-text');


        let model;


        const loadModel = async () => {

            model = await mobilenet.load();

        };


        const classifyImage = async () => {

            resultDiv.classList.add('hidden');

            loadingText.classList.remove('hidden');


            const predictions = await model.classify(imagePreview);

            

            loadingText.classList.add('hidden');

            resultDiv.classList.remove('hidden');


            const prediction = predictions[0];

            const className = prediction.className;

            let displayMessage = `This is likely: ${className}`;

            

            const commonRecyclableClasses = {

                'plastic_bottle': 'Plastic Bottle ♻️',

                'water_bottle': 'Plastic Water Bottle ♻️',

                'carton': 'Cardboard Box ♻️',

                'paper_towel': 'Paper Waste ♻️',

                'cup': 'Cup 🗑️',

                'wine_bottle': 'Glass Bottle ♻️',

                'soda_bottle': 'Plastic Bottle ♻️',

                'garbage_can': 'Garbage Can 🗑️',

                'can_opener': 'Metal Can ♻️',

                'pop_bottle': 'Plastic Bottle ♻️',

            };


            if (commonRecyclableClasses[className.toLowerCase()]) {

                displayMessage = `This is a recognized recyclable item: ${commonRecyclableClasses[className.toLowerCase()]}`;

            }


            predictionText.textContent = displayMessage;

        };


        imageUpload.addEventListener('change', (event) => {

            const file = event.target.files[0];

            if (file) {

                const reader = new FileReader();

                reader.onload = (e) => {

                    imagePreview.src = e.target.result;

                    imageContainer.classList.remove('hidden');

                    classifyBtn.classList.remove('hidden');

                    resultDiv.classList.add('hidden');

                    loadingText.classList.add('hidden');

                };

                reader.readAsDataURL(file);

            }

        });


        classifyBtn.addEventListener('click', classifyImage);


        window.onload = loadModel;

    </script>
